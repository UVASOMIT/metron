String Extensions

String.lower() 
              
              var name = "JOHN";
              name.lower();
              console.log(name);
              //result: "john"
String.upper() 
              
              var name = "john";
              name.upper();
              console.log(name);
              //result: "JOHN"
String.trim() 
              
              var name = "   John   ";
              name.trim();
              console.log(name);
              //result: "John"
String.ltrim() 
              
              var name = "   John   ";
              name.ltrim();
              console.log(name);
              //result: "John   "
String.rtrim() 
              
              var name = "   John   ";
              name.rtrim();
              console.log(name);
              //result: "   John"
String.normalize() 
              
              var para = "   John  Smith  from          New  Jersey  ";
              para.normalize();
              console.log(para);
              //result: "John Smith from New Jersey"
String.startsWith() 
              
              var name = "John";
              console.log(name.startsWith("J"));
              //result: true
String.endsWith() 
              
              var name = "John";
              console.log(name.endsWith("J"));
              //result: false
String.capFirst() 
              
              var name = "john";
              name.capFirst();
              console.log(name);
              //result: "John"
String.capWords() 
              
              var para = "john smith from the pinelands";
              para.capWords();
              console.log(para);
              //result: "John Smith From The Pinelands"
String.truncateWords() 
              
              var para = "John Smith from the Pinelands.";
              para.truncateWords(2);
              console.log(para);
              //result: "John Smith"
String.truncateWordsWithHtml() 
              
              var para = "<strong>John Smith</strong> from the Pinelands.";
              para.truncateWordsWithHtml(1);
              console.log(para);
              //result: "<strong>John</strong>"
